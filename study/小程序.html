二.自定义组件

  1. 在根目录下建立 components 文件夹
  2. 在components文件夹下建立组件的相应文件,例如components/my-button/my-button.wxml

在组件的wxml中书写模板

<view class='title-ch'>{{titleCh}}</view>
<view class='line' style="background: linear-gradient(to right, {{bgColor}})"></view>
<view class='title-en'>{{titleEn}}</view>

在组件的js中传入需要的属性 例1

properties: {
    titleCh: {
      type: String,
      value: ""
    },
    bgColor: {
      type: String,
      value: ""
    },
    titleEn: {
      type: String,
      value: ""
    }
  },

  1. 配置要引入的页面的json:

    "usingComponents": {
 "titleBlock": "/components/titleBlock/titleBlock"
}

  2. 在要引入的页面wxml中传值并调用

    <titleBlock titleCh="关于我们" bgColor="#C000FF,#00C7EF" titleEn="About us"></titleBlock>

四.选择图片

chooseImg() {
    var that = this;
    wx.chooseImage({
      count: 3,
      sizeType: ['original', 'compressed'],
      sourceType: ['album', 'camera'],
      success(res) {
        console.log(res);
        // tempFilePath可以作为img标签的src属性显示图片
        const tempFilePaths = res.tempFilePaths;
        that.setData({
          imgLs: tempFilePaths
        })
      }
    })
  }

五.上传图片

uploadImg() {
    var that = this;
    wx.uploadFile({
      url: 'https://sm.ms/api/upload', //仅为示例，非真实的接口地址
      filePath: that.data.imgLs[0],
      name: 'smfile',
      success(res) {
        const data = JSON.parse(res.data);
        console.log(data);
        that.setData({
          uploadedImgSrc: data.data.url
        })
      }
    })
  }

八,获取用户信息

<button wx:if="{{canIUse}}" open-type="getUserInfo" bindgetuserinfo="bindGetUserInfo">授权登录</button>
<view wx:else>请升级微信版本</view>

Page({
  data: {
    canIUse: wx.canIUse('button.open-type.getUserInfo')
  },
  onLoad: function() {
    // 查看是否授权
    wx.getSetting({
      success: function(res){
        if (res.authSetting['scope.userInfo']) {
          // 已经授权，可以直接调用 getUserInfo 获取头像昵称
          wx.getUserInfo({
            success: function(res) {
              console.log(res.userInfo)
            }
          })
        }
      }
    })
  },
  bindGetUserInfo: function(e) {
    console.log(e.detail.userInfo)
  }
})
